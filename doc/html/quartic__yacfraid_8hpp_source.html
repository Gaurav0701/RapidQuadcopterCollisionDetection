30  inline size_t
31  yacfraidQuarticSolve(const double& a, const double& b, const double& c, const double& d,
32  double& root1, double& root2, double& root3, double& root4)
33  {
34  int j;
35  double y;
36  double v3[4];
37  double det0,det1,det2,det3;
38  double det0rt,det1rt,det2rt,det3rt;
39  double e,f,g,h,k;
40  double fsq,gsq,hsq,invk;
41  double P,Q,R,U;
42 
43  double rts[4];
44  double worst3[3];
45  double qrts[4][3]; /* quartic roots for each cubic root */
46 
47  if (d == 0.0)
48  {
49  root1 = 0.0;
50  return cubicSolve(a,b,c,root2,root3,root4) + 1;
51  }
52 
53  double asq = a * a;
54  double acu = a * asq;
55  double b4 = b * 4.0;
56  size_t n3 = 0;
57  int n4[4];
58 
59  P = asq * b - b4 * b + 2.0 * a * c + 16.0 * d ;
60  Q = asq * c - b4 * c + 8.0 * a * d;
61  R = asq * d - c * c;
62  U = acu - b4 * a + 8.0 * c;
63  n4[0] = 0;
64 
65 
66  asq = a*a;
67  acu = a*asq;
68  b4 = b*4.0;
69  n3 = 0;
70 
71  P = asq*b - b4*b + 2.0*a*c + 16.0*d ;
72  Q = asq*c - b4*c + 8.0*a*d;
73  R = asq*d - c*c ;
74  U = acu - b4*a + 8.0*c;
75  n4[0] = 0;
76  if (U == 0.0)
77  {
78  if (P == 0.0)
79  {
80  det0 = 3.0*asq - 8.0*b;
81  if (det0 < 0.0)
82  goto done;
83  det0rt = sqrt(det0);
84  qrts[0][0] = (-a + det0rt)*0.25;
85  qrts[1][0] = qrts[0][0];
86  qrts[2][0] = (-a - det0rt)*0.25;
87  qrts[3][0] = qrts[2][0];
88  n4[0] = 4;
89  goto done;
90  } /* P=0 */
91  else
92  {
93  det1 = asq*asq - 8.0*asq*b + 16.0*b*b - 64.0*d;
94  if (det1 < 0.0)
95  goto done;;
96  n4[0] = 0;
97  det1rt = sqrt(det1);
98  det2 = 3.0*asq - 8.0*b + 2.0*det1rt;
99  if (det2 >= 0.0)
100  {
101  det2rt = sqrt(det2);
102  qrts[0][0] = (-a + det2rt)*0.25;
103  qrts[1][0] = (-a - det2rt)*0.25;
104  n4[0] = 2;
105  }
106  det3 = 3.0*asq - 8.0*b - 2.0*det1rt;
107  if (det3 >= 0.0)
108  {
109  det3rt = sqrt(det3);
110  qrts[n4[0]++][0] = (-a + det3rt)*0.25;
111  qrts[n4[0]++][0] = (-a - det3rt)*0.25;
112  }
113  goto done;
114  } /* P<>0 */
115  }
116 
117  n3 = cubicSolve(P/U,Q/U,R/U,v3[0], v3[1], v3[2]);
118  for (size_t j3 = 0; j3 < n3; ++j3)
119  {
120  y = v3[j3];
121  j = 0;
122  k = a + 4.0*y;
123  if (k == 0.0)
124  goto donej3;
125  invk = 1.0/k;
126  e = (acu - 4.0*c - 2.0*a*b + (6.0*asq - 16.0*b)*y)*invk;
127  fsq = (acu + 8.0*c - 4.0*a*b)*invk;
128  if (fsq < 0.0)
129  goto donej3;
130  f = sqrt(fsq);
131  gsq = 2.0*(e + f*k);
132  hsq = 2.0*(e - f*k);
133  if (gsq >= 0.0)
134  {
135  g = sqrt(gsq);
136  qrts[j++][j3] = (-a - f - g)*0.25;
137  qrts[j++][j3] = (-a - f + g)*0.25;
138  }
139  if (hsq >= 0.0)
140  {
141  h = sqrt(hsq);
142  qrts[j++][j3] = (-a + f - h)*0.25;
143  qrts[j++][j3] = (-a + f + h)*0.25;
144  }
145  donej3:
146  n4[j3] = j;
147  for (j = 0; j < n4[j3]; ++j)
148  rts[j] = qrts[j][j3];
149 
150  worst3[j3] = quarticError(a, b, c, d, rts, n4[j3]);
151  } /* j3 loop */
152  done:
153  size_t j3 = 0;
154  if (n3 > 1)
155  {
156  if ((n4[1] > n4[j3]) ||
157  ((worst3[1] < worst3[j3] ) && (n4[1] == n4[j3]))) j3 = 1;
158 
159  if ((n4[2] > n4[j3]) ||
160  ((worst3[2] < worst3[j3] ) && (n4[2] == n4[j3]))) j3 = 2;
161  }
162 
163  root1 = qrts[0][j3];
164  root2 = qrts[1][j3];
165  root3 = qrts[2][j3];
166  root4 = qrts[3][j3];
167 
168 
169  return (n4[j3]);
170  }
171  }
