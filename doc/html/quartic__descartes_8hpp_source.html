29  inline size_t
30  descartesQuarticSolve(const double& a, const double& b, const double& c, const double& d,
31  double& root1, double& root2, double& root3, double& root4)
32  {
33  double rts[4];
34  double worst3[3];
35  double qrts[4][3]; /* quartic roots for each cubic root */
36 
37  if (d == 0.0)
38  {
39  root1 = 0.0;
40  return cubicSolve(a,b,c,root2,root3,root4) + 1;
41  }
42 
43  int j, n4[4];
44  double v1[4] = {0,0,0,0},
45  v2[4] = {0,0,0,0},v3[4]={0,0,0,0};
46  double k,y;
47  double p,q,r;
48  double e0,e1,e2;
49  double g,h;
50  double asq;
51  double ainv4;
52  double e1invk;
53 
54  asq = a*a;
55  e2 = b - asq * (3.0/8.0);
56  e1 = c + a*(asq*0.125 - b*0.5);
57  e0 = d + asq*(b*0.0625 - asq*(3.0/256.0)) - a*c*0.25;
58 
59  p = 2.0*e2;
60  q = e2*e2 - 4.0*e0;
61  r = -e1*e1;
62 
63  size_t n3 = cubicSolve(p,q,r,v3[0],v3[1],v3[2]);
64  for (size_t j3 = 0; j3 < n3; ++j3)
65  {
66  y = v3[j3];
67  if (y <= 0.0)
68  n4[j3] = 0;
69  else
70  {
71  k = std::sqrt(y);
72  ainv4 = a*0.25;
73  e1invk = e1/k;
74  g = (y + e2 + e1invk)*0.5;
75  h = (y + e2 - e1invk)*0.5 ;
76  bool n1 = quadraticSolve( g, -k, 1.0, v1[0], v1[1]);
77  bool n2 = quadraticSolve( h, k, 1.0, v2[0], v2[1]);
78  qrts[0][j3] = v1[0] - ainv4;
79  qrts[1][j3] = v1[1] - ainv4;
80  qrts[n1*2][j3] = v2[0] - ainv4;
81  qrts[n1*2+1][j3] = v2[1] - ainv4;
82  n4[j3]= n1*2 + n2*2;
83  } /* y>=0 */
84  for (j = 0; j < n4[j3]; ++j)
85  rts[j] = qrts[j][j3];
86  worst3[j3] = quarticError(a, b, c, d, rts, n4[j3]);
87  } /* j3 loop */
88  size_t j3 = 0;
89  if (n3 != 1)
90  {
91  if ((n4[1] > n4[j3]) ||
92  ((worst3[1] < worst3[j3] ) && (n4[1] == n4[j3]))) j3 = 1;
93  if ((n4[2] > n4[j3]) ||
94  ((worst3[2] < worst3[j3] ) && (n4[2] == n4[j3]))) j3 = 2;
95  }
96 
97  root1 = qrts[0][j3];
98  root2 = qrts[1][j3];
99  root3 = qrts[2][j3];
100  root4 = qrts[3][j3];
101 
102  return (n4[j3]);
103  }
104  }
