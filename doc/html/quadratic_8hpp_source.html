22  //Solve a quadratic of the form x^2 + B x + C == 0
23  inline bool quadraticSolve(const double& B, const double& C,
24  double& root1, double& root2)
25  {
26  double discriminant = (B * B) - (4 * C);
27 
28  //Cannot do imaginary numbers, yet
29  if (discriminant < 0) return false;
30 
31  //This avoids a cancellation of errors. See
32  //http://en.wikipedia.org/wiki/Quadratic_equation#Floating_point_implementation
33  double t = -0.5 * ( B + ((B < 0) ? -1 : 1) * std::sqrt(discriminant));
34 
35  root1 = t;
36  if(t==0) root2 = 0;
37  else root2 = C / t;
38 
39  return true;
40  }

43  inline bool quadSolve(const double& C, const double& B, const double& A,
44  double& root1, double& root2)
45  {
46  // Contingency: if A = 0, not a quadratic = linear
47  if(A == 0)
48  {
49  //If B is zero then we have a NaN
50  if(B == 0) return false;
51 
52  root1 = -1.0 * C / B;
53  root2 = root1;
54  }
55 
56  double discriminant = (B * B) - (4 * A * C);
57 
58  //Cannot do imaginary numbers, yet
59  if (discriminant < 0) return false;
60 
61  //This avoids a cancellation of errors. See
62  //http://en.wikipedia.org/wiki/Quadratic_equation#Floating_point_implementation
63  double t = -0.5 * ( B + ((B < 0) ? -1 : 1) * std::sqrt(discriminant));
64 
65  root1 = t / A;
66  root2 = C / t;
67 
68  return true;
69  }

71  typedef enum {
72  ROOT_SMALLEST_EITHER = 1,
73  ROOT_SMALLEST_POSITIVE = 2,
74  ROOT_SMALLEST_NEGATIVE = 4,
75  ROOT_LARGEST_EITHER = 8,
76  ROOT_LARGEST_POSITIVE = 16,
77  ROOT_LARGEST_NEGATIVE = 32
78  } rootTypeEnum;

80  template<rootTypeEnum rootType>
81  inline bool quadSolve(const double& C, const double& B, const double& A, double& ans)
82  {
83  double root1(0), root2(0);
84 
85  if (!quadSolve(C,B,A,root1,root2)) return false;
86 
87  switch (rootType)
88  {
89  case ROOT_SMALLEST_EITHER:
90  ans = (fabs(root1) < fabs(root2)) ? root1 : root2;
91  break;
92  case ROOT_LARGEST_EITHER:
93  ans = (fabs(root1) < fabs(root2)) ? root2 : root1;
94  case ROOT_LARGEST_NEGATIVE:
95  if (root1 < 0 && root2 < 0)
96  ans = ((root1 < root2) ? root1 : root2);
97  else if (root1 < 0)
98  ans = root1;
99  else if (root2 < 0)
100  ans = root2;
101  else
102  return false;
103  break;
104  case ROOT_SMALLEST_NEGATIVE:
105  if (root1 < 0 && root2 < 0)
106  ans = ((root1 < root2) ? root2 : root1);
107  else if (root1 < 0)
108  ans = root1;
109  else if (root2 < 0)
110  ans = root2;
111  else
112  return false;
113  break;
114  case ROOT_LARGEST_POSITIVE:
115  if (root1 > 0 && root2 > 0)
116  ans = ((root1 > root2) ? root1 : root2);
117  else if (root1 > 0)
118  ans = root1;
119  else if (root2 > 0)
120  ans = root2;
121  else
122  return false;
123  break;
124  case ROOT_SMALLEST_POSITIVE:
125  if (root1 > 0 && root2 > 0)
126  ans = ((root1 > root2) ? root2 : root1);
127  else if (root1 > 0)
128  ans = root1;
129  else if (root2 > 0)
130  ans = root2;
131  else
132  return false;
133  break;
134  default:
135  throw std::runtime_error("Unknown root selected");
136  break;
137  }
138  return true;
139  }
