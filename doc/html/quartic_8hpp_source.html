32  inline void quarticNewtonRootPolish(const double& a, const double& b, const double& c, const double& d,
33  double& root, size_t iterations)
34  {
35  for (size_t it = 0; it < iterations; ++it)
36  {
37  double error = (((root + a)*root + b) * root + c) * root + d;
38  double derivative = ((4 * root + 3 * a) * root + 2 * b) * root + c;
39 
40  if ((error == 0) || derivative == 0) return;
41 
42  root -= error / derivative;
43  }
44  }
45  }

47  //Solves quartics of the form x^4 + a x^3 + b x^2 + c x + d ==0
48  inline size_t quarticSolve(const double& a, const double& b, const double& c, const double& d,
49  double& root1, double& root2, double& root3, double& root4)
50  {
51  static const double maxSqrt = std::sqrt(std::numeric_limits<double>::max());
52 
53  if (std::abs(a) > maxSqrt)
54  yacfraidQuarticSolve(a,b,c,d,root1,root2,root3,root4);
55 
56  if (d == 0)
57  {//Solve a cubic with a trivial root of 0
58  root1 = 0;
59  return 1 + cubicSolve(a, b, c, root2, root3, root4);
60  }
61 
62  if ((a == 0) && (c== 0))
63  {//We have a biquadratic
64  double quadRoot1,quadRoot2;
65  if (quadraticSolve(d,b,1, quadRoot1, quadRoot2))
66  {
67  if (quadRoot1 < quadRoot2) std::swap(quadRoot1,quadRoot2);
68 
69  if (quadRoot1 < 0)
70  return 0;
71 
72  root1 = std::sqrt(quadRoot1);
73  root2 = -std::sqrt(quadRoot1);
74 
75  if (quadRoot2 < 0)
76  return 2;
77 
78  root3 = std::sqrt(quadRoot2);
79  root4 = -std::sqrt(quadRoot2);
80  return 4;
81  }
82  else
83  return 0;
84  }
85 
86  //Now we have to resort to some dodgy formulae!
87  size_t k = 0, nr;
88  if (a < 0.0) k += 2;
89  if (b < 0.0) k += 1;
90  if (c < 0.0) k += 8;
91  if (d < 0.0) k += 4;
92  switch (k)
93  {
94  case 3 :
95  case 9 :
96  nr = ferrariQuarticSolve(a,b,c,d,root1,root2,root3,root4);
97  break;
98  case 5 :
99  nr = descartesQuarticSolve(a,b,c,d,root1,root2,root3,root4);
100  break;
101  case 15 :
102  //This algorithm is stable if we flip the sign of the roots
103  nr = descartesQuarticSolve(-a,b,-c,d,root1,root2,root3,root4);
104  root1 *=-1; root2 *=-1; root3 *=-1; root4 *=-1;
105  break;
106  default:
107  nr = neumarkQuarticSolve(a,b,c,d,root1,root2,root3,root4);
108  //nr = ferrariQuarticSolve(a,b,c,d,root1,root2,root3,root4);
109  //nr = yacfraidQuarticSolve(a,b,c,d,root1,root2,root3,root4);
110  break;
111  }
112 
113  if (nr) detail::quarticNewtonRootPolish(a, b, c, d, root1, 15);
114  if (nr>1) detail::quarticNewtonRootPolish(a, b, c, d, root2, 15);
115  if (nr>2) detail::quarticNewtonRootPolish(a, b, c, d, root3, 15);
116  if (nr>3) detail::quarticNewtonRootPolish(a, b, c, d, root4, 15);
117 
118  return nr;
119  }
